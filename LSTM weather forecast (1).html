#!/usr/bin/env python # coding: utf-8 import numpy as np import pandas as pd import tensorflow as tf import matplotlib.pyplot as plt from sklearn.preprocessing import MinMaxScaler from tensorflow.keras.layers import LSTM, GRU, Dense # Load and filter dataset df = pd.read_csv('city_temperature.csv') df = df[df['Year'] == 1996] newdf = df[['City', 'AvgTemperature']] newdf = newdf[newdf['City'] == 'Algiers'] # Convert temperature to Celsius and round newdf['AvgTemperature'] = newdf['AvgTemperature'].apply(lambda x: round((5 / 9) * (x - 32), 1)) # Clean DataFrame index newdf.reset_index(drop=True, inplace=True) # Extract temperature data temps = newdf["AvgTemperature"].values.astype("float32") # Detect and fix outliers using IQR method temps_series = newdf["AvgTemperature"].copy() Q1 = temps_series.quantile(0.25) Q3 = temps_series.quantile(0.75) IQR = Q3 - Q1 lower_bound = Q1 - 1.5 * IQR upper_bound = Q3 + 1.5 * IQR temps_series = temps_series.where((temps_series >= lower_bound) & (temps_series <= upper_bound), np.nan) temps_series = temps_series.fillna(temps_series.rolling(window=3, center=True).mean()) temps_series = temps_series.fillna(method='bfill').fillna(method='ffill') temps = temps_series.values.astype("float32") # Normalize data scaler = MinMaxScaler() temps_scaled = scaler.fit_transform(temps.reshape(-1, 1)).flatten() # Create sequences sequence_length = 2 X, y = [], [] for i in range(len(temps_scaled) - sequence_length): X.append(temps_scaled[i:i + sequence_length]) y.append(temps_scaled[i + sequence_length]) X = np.array(X).reshape((len(X), sequence_length, 1)) y = np.array(y) # Train/test split split_index = int(len(X) * 0.8) X_train, X_test = X[:split_index], X[split_index:] y_train, y_test = y[:split_index], y[split_index:] # Define and train LSTM model model = tf.keras.Sequential([ tf.keras.layers.InputLayer(shape=(sequence_length, 1)), tf.keras.layers.LSTM(50, activation='relu'), tf.keras.layers.Dense(1) ]) model.compile(optimizer='adam', loss='mse') model.fit(X_train, y_train, epochs=100, verbose=1, validation_split=0.1) # Predict and plot y_pred = model.predict(X_test) y_pred_rescaled = scaler.inverse_transform(y_pred) y_test_rescaled = scaler.inverse_transform(y_test.reshape(-1, 1)) plt.figure(figsize=(10, 4)) plt.plot(y_test_rescaled, label="True Temp") plt.plot(y_pred_rescaled, label="Predicted Temp") plt.xlabel("Days") plt.ylabel("Avg Temperature (째C)") plt.title("LSTM Weather Forecast - Algiers") plt.legend() plt.grid(True) plt.tight_layout() plt.show() # Predict next day last_seq = temps_scaled[-sequence_length:].reshape((1, sequence_length, 1)) next_day_pred_scaled = model.predict(last_seq) next_day_pred = scaler.inverse_transform(next_day_pred_scaled) print("Predicted next day's temperature:", round(next_day_pred[0][0], 2), "째C") # Define and train stacked LSTM model model = tf.keras.Sequential([ tf.keras.layers.InputLayer(shape=(sequence_length, 1)), tf.keras.layers.LSTM(64, return_sequences=True), tf.keras.layers.LSTM(32), tf.keras.layers.Dense(1) ]) model.compile(optimizer='adam', loss='mse') model.fit(X_train, y_train, epochs=100, verbose=1, validation_split=0.1) # Predict and plot for stacked LSTM y_pred = model.predict(X_test) y_pred_rescaled = scaler.inverse_transform(y_pred) plt.figure(figsize=(10, 4)) plt.plot(y_test_rescaled, label="True Temp") plt.plot(y_pred_rescaled, label="Predicted Temp") plt.xlabel("Days") plt.ylabel("Avg Temperature (째C)") plt.title("Stacked LSTM Weather Forecast - Algiers") plt.legend() plt.grid(True) plt.tight_layout() plt.show() # Predict next day last_seq = temps_scaled[-sequence_length:].reshape((1, sequence_length, 1)) next_day_pred_scaled = model.predict(last_seq) next_day_pred = scaler.inverse_transform(next_day_pred_scaled) print("Predicted next day's temperature:", round(next_day_pred[0][0], 2), "째C")
